<script>
$(function() {
    var pcSearchTimeout;
    var $pcSearchInput = $('#sch_stx');
    var $pcSearchPanel = $('#pcSearchPanel');
    var $pcSearchDropdown = $('#pcSearchDropdown');
    var $pcSearchClear = $('#pc_sch_clear');

    // ========== LocalStorage 관리 함수 ==========

    // 최근 검색어 가져오기
    function getRecentKeywords() {
        try {
            var data = localStorage.getItem('recentSearchKeywords');
            return data ? JSON.parse(data) : [];
        } catch (e) {
            console.error('LocalStorage 읽기 오류:', e);
            return [];
        }
    }

    // 최근 검색어 저장
    function saveRecentKeyword(keyword) {
        if (!keyword || keyword.trim() === '') return;

        try {
            var keywords = getRecentKeywords();
            var trimmedKeyword = keyword.trim();

            // 중복 제거 (이미 있으면 제거)
            keywords = keywords.filter(function(item) {
                return item.keyword !== trimmedKeyword;
            });

            // 맨 앞에 추가
            keywords.unshift({
                keyword: trimmedKeyword,
                timestamp: Date.now()
            });

            // 최대 30개까지만 유지
            if (keywords.length > 30) {
                keywords = keywords.slice(0, 30);
            }

            localStorage.setItem('recentSearchKeywords', JSON.stringify(keywords));
        } catch (e) {
            console.error('LocalStorage 저장 오류:', e);
        }
    }

    // 최근 검색어 개별 삭제
    function removeRecentKeyword(keyword) {
        try {
            var keywords = getRecentKeywords();
            keywords = keywords.filter(function(item) {
                return item.keyword !== keyword;
            });
            localStorage.setItem('recentSearchKeywords', JSON.stringify(keywords));
            displayRecentKeywords();
        } catch (e) {
            console.error('LocalStorage 삭제 오류:', e);
        }
    }

    // 최근 검색어 전체 삭제
    function clearAllRecentKeywords() {
        try {
            localStorage.removeItem('recentSearchKeywords');
            displayRecentKeywords();
        } catch (e) {
            console.error('LocalStorage 전체 삭제 오류:', e);
        }
    }

    // ========== 표시 함수 ==========

    // 최근 검색어 표시
    function displayRecentKeywords() {
        var keywords = getRecentKeywords();
        var $list = $('#pcSearchRecentList');
        var $moreBtn = $('#pcSearchRecentMoreBtn');

        $list.empty();

        if (keywords.length === 0) {
            $list.html('<li class="pc-search-empty">최근 검색 내역이 없습니다.</li>');
            $list.removeClass('collapsed');
            $moreBtn.hide();
            return;
        }

        // 초기에는 collapsed 상태로 설정 (5개까지만 표시)
        if (keywords.length > 5) {
            $list.addClass('collapsed');
            $moreBtn.show().removeClass('expanded');
            $moreBtn.find('.more-text').text('펼쳐서 더보기');
        } else {
            $list.removeClass('collapsed');
            $moreBtn.hide();
        }

        keywords.forEach(function(item) {
            var html = '<li class="pc-search-recent-item">';
            html += '<div class="pc-search-recent-keyword" data-keyword="' + item.keyword + '">';
            html += '<svg class="pc-search-recent-icon" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">';
            html += '<circle cx="12" cy="12" r="10"/><path d="M12 6v6l4 2"/>';
            html += '</svg>';
            html += '<span>' + item.keyword + '</span>';
            html += '</div>';
            html += '<button type="button" class="pc-search-recent-delete" data-keyword="' + item.keyword + '">×</button>';
            html += '</li>';
            $list.append(html);
        });
    }

    // 인기 검색어 로드 및 표시
    function loadPopularKeywords() {
        $.ajax({
            url: '/popular_keyword_api.php',
            type: 'GET',
            dataType: 'json',
            success: function(res) {
                var $list = $('#pcSearchPopularList');
                var $moreBtn = $('#pcSearchPopularMoreBtn');

                $list.empty();

                if (res.success && res.data.length > 0) {
                    // 초기에는 collapsed 상태로 설정 (5개까지만 표시)
                    if (res.data.length > 5) {
                        $list.addClass('collapsed');
                        $moreBtn.show().removeClass('expanded');
                        $moreBtn.find('.more-text').text('펼쳐서 더보기');
                    } else {
                        $list.removeClass('collapsed');
                        $moreBtn.hide();
                    }

                    res.data.forEach(function(item, index) {
                        var html = '<li class="pc-search-popular-item" data-pk-id="' + item.pk_id + '" data-keyword="' + item.pk_keyword + '" data-link-url="' + (item.pk_link_url || '') + '">';
                        html += '<span class="pc-search-popular-rank">' + (index + 1) + '</span>';
                        html += '<span class="pc-search-popular-keyword">' + item.pk_keyword + '</span>';
                        html += '</li>';
                        $list.append(html);
                    });
                } else {
                    $list.html('<li class="pc-search-empty">등록된 인기 검색어가 없습니다.</li>');
                    $list.removeClass('collapsed');
                    $moreBtn.hide();
                }
            },
            error: function() {
                $('#pcSearchPopularList').html('<li class="pc-search-empty">인기 검색어를 불러올 수 없습니다.</li>');
                $('#pcSearchPopularList').removeClass('collapsed');
                $('#pcSearchPopularMoreBtn').hide();
            }
        });
    }

    // 최근 검색어 + 인기 검색어 로드
    function loadRecentAndPopularKeywords() {
        displayRecentKeywords();
        loadPopularKeywords();
    }

    // ========== 이벤트 핸들러 ==========

    // 드롭다운 위치 및 크기 계산 함수
    function updateDropdownPosition() {
        var panelOffset = $pcSearchPanel.offset();
        var panelWidth = $pcSearchPanel.outerWidth();
        var panelHeight = $pcSearchPanel.outerHeight();

        // 60% 더 넓게 (1.6배)
        var dropdownWidth = panelWidth * 1.6;

        // 중앙 정렬을 위한 left 계산
        var dropdownLeft = panelOffset.left - ((dropdownWidth - panelWidth) / 2);

        $pcSearchDropdown.css({
            'top': (panelOffset.top + panelHeight + 8) + 'px',
            'left': dropdownLeft + 'px',
            'width': dropdownWidth + 'px'
        });
    }

    // 검색창 포커스 시 드롭다운 표시
    $pcSearchInput.on('focus', function() {
        $pcSearchPanel.addClass('focused');
        updateDropdownPosition();
        $pcSearchDropdown.show();
        loadRecentAndPopularKeywords();
    });

    // 검색어 입력 이벤트
    $pcSearchInput.on('keyup input', function() {
        clearTimeout(pcSearchTimeout);
        var keyword = $(this).val().trim();

        // 삭제 버튼 표시/숨김
        if (keyword.length > 0) {
            $pcSearchClear.addClass('show');
        } else {
            $pcSearchClear.removeClass('show');
        }

        if (keyword.length === 0) {
            $('#pcSearchDefaultSection').show();
            $('#pcSearchAutocompleteSection').hide();
            return;
        }
